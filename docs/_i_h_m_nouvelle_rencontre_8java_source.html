246  private void gererHandler()
247  {
248  Log.d(TAG,"gererHandler()");
249  this.handler = new Handler(this.getMainLooper())
250  {
251  @Override
252  public void handleMessage(@NonNull Message message)
253  {
254  Log.d(TAG, "[Handler] id du message = " + message.what);
255  Log.d(TAG, "[Handler] contenu du message = " + message.obj.toString());
256 
257  switch (message.what)
258  {
259  case PeripheriqueBluetooth.CODE_CREATION_SOCKET:
260  Log.d(TAG, "[Handler] CREATION_SOCKET = " + message.obj.toString());
261  break;
262  case PeripheriqueBluetooth.CODE_CONNEXION_SOCKET:
263  Log.d(TAG, "[Handler] CODE_CONNEXION_SOCKET = " + message.obj.toString());
264  if(listeJoueurs.getCheckedItemCount() == NB_JOUEURS && peripheriqueBluetooth.estConnecte())
265  boutonLancerRencontre.setEnabled(true);
266  break;
267  case PeripheriqueBluetooth.CODE_DECONNEXION_SOCKET:
268  Log.d(TAG, "[Handler] DECONNEXION_SOCKET = " + message.obj.toString());
269  boutonLancerRencontre.setEnabled(false);
270  break;
271  case PeripheriqueBluetooth.CODE_RECEPTION_TRAME:
272  Log.d(TAG, "[Handler] RECEPTION_TRAME = " + message.obj.toString());
273  break;
274  }
275  }
276  };
277  }
