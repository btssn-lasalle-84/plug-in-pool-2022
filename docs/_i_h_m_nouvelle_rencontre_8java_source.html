234  {
235  @Override
236  public void handleMessage(@NonNull Message message)
237  {
238  Log.d(TAG, "[Handler] id du message = " + message.what);
239  Log.d(TAG, "[Handler] contenu du message = " + message.obj.toString());
240 
241  switch (message.what)
242  {
243  case PeripheriqueBluetooth.CODE_CREATION_SOCKET:
244  Log.d(TAG, "[Handler] CREATION_SOCKET = " + message.obj.toString());
245  break;
246  case PeripheriqueBluetooth.CODE_CONNEXION_SOCKET:
247  Log.d(TAG, "[Handler] CODE_CONNEXION_SOCKET = " + message.obj.toString());
248  if(listeJoueurs.getCheckedItemCount() == NB_JOUEURS && peripheriqueBluetooth.estConnecte())
249  boutonLancerRencontre.setEnabled(true);
250  break;
251  case PeripheriqueBluetooth.CODE_DECONNEXION_SOCKET:
252  Log.d(TAG, "[Handler] DECONNEXION_SOCKET = " + message.obj.toString());
253  boutonLancerRencontre.setEnabled(false);
254  break;
255  case PeripheriqueBluetooth.CODE_RECEPTION_TRAME:
256  Log.d(TAG, "[Handler] RECEPTION_TRAME = " + message.obj.toString());
257  break;
258  }
259  }
260  };
261  }
