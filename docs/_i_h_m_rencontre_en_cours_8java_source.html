293  @Override
294  public void handleMessage(@NonNull Message message)
295  {
296  Log.d(TAG, "[Handler] id du message = " + message.what);

298 
299  switch (message.what)

309  Log.d(TAG, "[Handler] DECONNEXION_SOCKET = " + message.obj.toString());
310  break;
311  case PeripheriqueBluetooth.CODE_RECEPTION_TRAME:
312  Log.d(TAG, "[Handler] RECEPTION_TRAME = " + message.obj.toString());
313  gererMessage(message.obj.toString());
314  break;
315  }
316  }
317  };

323  private void gererMessage(String message)
324  {
325  // format général : $PLUG;{TYPE};{DONNEES;}\r\n
326  // debug
327  String[] champs = message.split(Protocole.delimiteurChamp);
328  for(int i = 0; i < champs.length; i++)
329  {
330  Log.v(TAG, "[gererMessage] champs[" + i + "] = " + champs[i]);
331  }
332 
333  // vérification
334  if(!champs[Protocole.CHAMP_ENTETE_TRAME].equals(Protocole.delimiteurDebut))
335  {
336  Log.d(TAG, "[gererMessage] Erreur : en tête invalide !");
337  return;
338  }
339 
340  switch (champs[Protocole.CHAMP_TYPE_TRAME])
341  {
342  case Protocole.EMPOCHE:
343  // $PLUG;EMPOCHE;{COULEUR};{BLOUSE};\r\n
344  Log.d(TAG, "Trame EMPOCHE : Couleur = " + champs[Protocole.CHAMP_COULEUR] + " -> Blouse = " + champs[Protocole.CHAMP_BLOUSE]);
345  rencontre.stockerCoup(champs[Protocole.CHAMP_COULEUR], champs[Protocole.CHAMP_BLOUSE]);
346  if(!estPremierJoueurChoisi)
347  {
348  estPremierJoueurChoisi = true;
349  afficherListePremierJoueur(champs);
350  }
351  rencontre.jouerCoup();
352  actualiserScores(champs);
353  terminerRencontre();
354  break;
355  case Protocole.FAUTE:
356  // $PLUG;FAUTE;{COULEUR};{BLOUSE};\r\n
357  Log.d(TAG, "Trame FAUTE : Couleur = " + champs[Protocole.CHAMP_COULEUR] + " -> Blouse = " + champs[Protocole.CHAMP_BLOUSE]);
358  rencontre.faute(champs[Protocole.CHAMP_COULEUR], champs[Protocole.CHAMP_BLOUSE]);
359  actualiserScores(champs);
360  break;
361  case Protocole.SUIVANT:
362  // $PLUG;NEXT;
363  changerJoueurIHM();
364  Log.d(TAG, "Trame NEXT");
365  break;
366  case Protocole.ACK:
367  //
368  Log.d(TAG, "Trame ACK");
369  break;
370  case Protocole.ERREUR:
371  //
372  Log.d(TAG, "Trame ERREUR");
373  peripheriqueBluetooth.envoyer(Protocole.trameArreter);
374  break;
375  }
376  }
